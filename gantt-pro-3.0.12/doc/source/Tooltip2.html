Ext.define('Gnt.patches.Tooltip', {
    extend : 'Sch.util.Patch',

    requires   : ['Ext.tip.ToolTip'],
    target     : 'Ext.tip.ToolTip',
    minVersion : '5.1.0',

    overrides : {
        onDocMouseDown  : function () {
            if (this.isDisabled()) { 
                return; 
            }
            this.callParent(arguments);
        }
    }
});
