Ext.define('Gnt.patches.CellEditor', {
    extend : 'Sch.util.Patch',

    requires   : ['Ext.grid.CellEditor'],
    target     : 'Ext.grid.CellEditor',
    minVersion : '5.1.0',

    overrides : {
        onHide : function () {
            this.restoreCell();
            this.superclass.onHide.apply(this, arguments);
        }
    }
});
